import type { APIRoute } from "astro";
import { db, ViewCount } from "astro:db";

export const prerender = false;

export const GET: APIRoute = async () => {
  try {
    const allCounts = await db
      .select({
        slug: ViewCount.slug,
        count: ViewCount.count,
      })
      .from(ViewCount);

    // Transform the array into an object with slug as keys
    const countsObject = allCounts.reduce((acc, item) => {
      acc[item.slug] = item.count;
      return acc;
    }, {} as Record<string, number>);

    return new Response(
      JSON.stringify(countsObject),
      {
        status: 200,
        headers: {
          "content-type": "application/json",
          "cache-control": "no-store, max-age=0",
        },
      }
    );
  } catch (error) {
    return new Response(
      JSON.stringify({ error: "Failed to fetch view counts" }),
      { status: 500, headers: { "content-type": "application/json" } }
    );
  }
};
